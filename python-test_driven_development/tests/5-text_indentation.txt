>>> from 5-text_indentation import text_indentation


# Test simple sans aucun caractère nécessitant un saut de ligne
>>> text_indentation("Holberton School")
Holberton School

# Test avec un point (.) nécessitant un saut de ligne
>>> text_indentation("Holberton.School")
Holberton.
<BLANKLINE>
School

# Test avec plusieurs caractères (. ? :) nécessitant des sauts de lignes
>>> text_indentation("Holberton. School? How are you: John")
Holberton.
<BLANKLINE>
School?
<BLANKLINE>
How are you:
<BLANKLINE>
John

# Test avec une chaîne vide
>>> text_indentation("")
<BLANKLINE>

# Test avec un type invalide (nombre au lieu de texte)
>>> text_indentation(12)
Traceback (most recent call last):
    ...
TypeError: text must be a string

# Test avec une entrée sans espaces inutiles
>>> text_indentation("This. Is a test: Without extra spaces.")
This.
<BLANKLINE>
Is a test:
<BLANKLINE>
Without extra spaces.
